Proposed DB schema
Copy and paste into QDBD for diagram https://app.quickdatabasediagrams.com/#/

Guilds
----
id PK in
guild_id int UNIQUE
name string

Channels
----
id PK in
channel_id int UNIQUE
guild_id int FK >- Guilds.id
name string
is_voice bool
msg_limit int NULL #messages the bot can send to the channel per command

Guild_Config # bot configuration per guild
----
id PK int
guild int UNIQUE FK - Guilds.id
msg_limit int=100 #default message limit for new channels
# notification channels
general int FK - Channels.id # general bot message notification sent here
admin int fk - Channels.id # bot messages for admins only
steam in fk - Channels.id # Send steam sale alerts

Steam
----
id PK int
user_id int UNIQUE #discord user id
steam64_id int UNIQUE
steam_vanity string NULL
notify bool=false #PM user wishlist sales

Music
----
id PK int
guild int UNIQUE FK - Guilds.guild_id #Cnnected guild that has enrolled in music
channel int FK - Channels.id #Voice channel we connect to in the guild

Playlist
----
id PK int
name string UNIQUE
created_by int #discord user id.
created_in int FK - Music.id
allowed_in int FK -< Music.id #guild admin can block playlist but playlist stays with user

Songs
----
id PK int
title string
location string UNIQUE #yt link or local file.

Songs_to_Playlist
----
song PK int FK - Songs.id
playlist PK int FK - Playlist.id
id int UNIQUE AUTOINCREMENT
