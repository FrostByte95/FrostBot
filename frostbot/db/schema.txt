Proposed DB schema
Copy and paste into QDBD for diagram https://app.quickdatabasediagrams.com/#/

Guilds
----
id PK int
guild_id int UNIQUE
name string

Channels
----
id PK in
channel_id int UNIQUE
guild_id int FK >- Guilds.id
name string
is_voice bool
msg_limit int NULL #messages the bot can send to the channel per command

Guild_Config # bot configuration per guild
----
id PK int
guild int UNIQUE FK - Guilds.id
msg_limit int=100 #default message limit for new channels
# notification channels - If Null notification is turned off
general int NULL FK - Channels.id # general bot message notification sent here
admin int NULL FK - Channels.id # bot messages for admins only
steam int NULL FK - Channels.id # Send steam sale alerts

Users
----
id PK int
user_id int UNIQUE
name string

User_to_Guild
----
user PK int FK - Users.id
guild PK int FK - Guilds.id
id int UNIQUE AUTOINCREMENT

Steam
----
id PK int
user_id int FK - Users.id #discord user id
steam64_id int UNIQUE
steam_vanity string NULL
notify bool=false #PM user wishlist sales

Music # Guilds enrolled in Music player and channel to connect to
----
id PK int
guild int UNIQUE FK - Guilds.id #Connected guild that has enrolled in music
channel int FK - Channels.id #Voice channel we connect to in the guild

Playlists
----
id PK int
name string UNIQUE
created_by int FK - Users.id #discord user id.
created_in int FK - Music.id

blocked_playlists
----
playlist PK int FK - Playlists.id
music PK int FK - Music.id
id int UNIQUE AUTOINCREMENT

Songs
----
id PK int
title string
location string UNIQUE #yt link or local file.

Song_to_Playlist
----
song PK int FK - Songs.id
playlist PK int FK - Playlists.id
id int UNIQUE AUTOINCREMENT
